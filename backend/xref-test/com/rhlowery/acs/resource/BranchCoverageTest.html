91      @Test
92      public void testAccessRequestResourceBranches() {
93          // 1. Authenticate
94          String token = given()
95              .contentType(ContentType.JSON)
96              .body(Map.of("userId", "admin", "password", "admin", "role", "ADMIN", "groups", java.util.List.of("admins")))
97              .post("/api/auth/login")
98              .then()
99              .statusCode(200)
100             .extract().cookie("bff_jwt");
101 
102         // 2. Submit a request
103         String id = java.util.UUID.randomUUID().toString();
104         given()
105             .auth().oauth2(token).cookie("bff_jwt", token)
106             .contentType(ContentType.JSON)
107             .body(java.util.List.of(Map.of(
108                 "id", id,
109                 "catalogName", "uc-oss",
110                 "tableName", "table1",
111                 "privileges", java.util.List.of("SELECT")
112             )))
113             .post("/api/storage/requests")
114             .then()
115             .statusCode(200);
116 
117         // 3. Verify non-approved (hits !"APPROVED".equals(r.status()))
118         given()
119             .auth().oauth2(token).cookie("bff_jwt", token)
120             .contentType(ContentType.JSON)
121             .post("/api/storage/requests/" + id + "/verify")
122             .then()
123             .statusCode(400);
124 
125         // 4. Approve it
126         given()
127             .auth().oauth2(token).cookie("bff_jwt", token)
128             .contentType(ContentType.JSON)
129             .post("/api/storage/requests/" + id + "/approve")
130             .then()
131             .statusCode(200);
132 
133         // 5. Verify it (hits verified flow)
134         given()
135             .auth().oauth2(token).cookie("bff_jwt", token)
136             .contentType(ContentType.JSON)
137             .post("/api/storage/requests/" + id + "/verify")
138             .then()
139             .statusCode(200);
140             
141         // 6. Drift test (hits verified == false)
142         // We need a way to mock drift. Since we use AbstractMockProvider, 
143         // if we change the policy in the mock directly, we can trigger drift.
144         // But for branch coverage, we just need to hit the 'else' block.
145         // Actually, without mocking the service it might be hard to trigger the else block 
146         // unless we have a specific catalog that always returns wrong permissions.
147     }
