10  @QuarkusTest
11  public class AuditResourceTest {
12  
13      @Test
14      public void testAuditLogging() {
15          // 1. Post audit log
16          given()
17              .contentType(ContentType.JSON)
18              .body(Map.of(
19                  "type", "ACCESS_GRANTED",
20                  "actor", "admin",
21                  "details", Map.of("resource", "table1")
22              ))
23              .post("/api/audit/log")
24              .then()
25              .statusCode(200)
26              .body("status", is("success"));
27  
28          // 2. Get audit logs
29          given()
30              .get("/api/audit/log")
31              .then()
32              .statusCode(200)
33              .body("$.size()", greaterThanOrEqualTo(1))
34              .body("[0].type", notNullValue());
35              
36          // 3. Post audit UI log
37          given()
38              .contentType(ContentType.JSON)
39              .body(Map.of("level", "INFO", "message", "test-ui"))
40              .post("/api/audit/log/ui")
41              .then()
42              .statusCode(204);
43      }
44
