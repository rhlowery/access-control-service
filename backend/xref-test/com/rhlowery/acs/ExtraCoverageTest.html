212     public void testAccessRequestBranches() {
213         String id1 = UUID.randomUUID().toString();
214         AccessRequest req1 = new AccessRequest(id1, "alice", "alice", "USER", "cat", "sch", "tbl", "TABLE", 
215             List.of("READ"), "PENDING", 0L, 0L, "J", null, List.of("group1"), Collections.emptyMap(), null);
216         
217         accessRequestService.saveRequests(List.of(req1), "alice", List.of(), false);
218         assertTrue(accessRequestService.getAllRequests("alice", List.of(), false).size() > 0);
219         assertTrue(accessRequestService.getAllRequests("bob", List.of("group1"), false).size() > 0);
220 
221         AccessRequest update = new AccessRequest(id1, "alice", "alice", "USER", "cat", "sch", "tbl", "TABLE", 
222             List.of("READ"), "APPROVED", 0L, 0L, "J", null, List.of("group1"), Collections.emptyMap(), null);
223         
224         accessRequestService.saveRequests(List.of(update), "admin", List.of("admins"), true);
225         
226         AccessRequest retrieved = accessRequestService.getRequestById(id1).orElseThrow();
227         assertEquals("APPROVED", retrieved.status());
228     }

230     @Test
231     public void testUserServiceBranches() {
232         userService.saveGroup(new com.rhlowery.acs.domain.Group("tg1", "N", "D", "P"));
233         userService.saveGroup(new com.rhlowery.acs.domain.Group("tg1", "N2", "D2", "P2")); // Hit update branch
234         userService.updateGroupPersona("tg1", "NEW_P");
235         assertEquals("NEW_P", userService.getGroup("tg1").get().persona());
236         
237         userService.saveUser(new com.rhlowery.acs.domain.User("tu1", "N", "E", "R", List.of("tg1"), "P"));
238         userService.saveUser(new com.rhlowery.acs.domain.User("tu1", "N2", "E2", "R2", List.of("tg1"), "P2")); // Hit update branch
239         userService.updateUserGroups("tu1", List.of("tg2"));
240         userService.updateUserPersona("tu1", "NEW_P");
241         
242         assertTrue(userService.listUsers().size() > 0);
243         assertTrue(userService.listGroups().size() > 0);
244         
245         userService.clear();
246         assertEquals(0, userService.listUsers().size());
247     }
