1   package com.rhlowery.acs;

3   import io.cucumber.java.en.Given;
4   import io.cucumber.java.en.Then;
5   import io.cucumber.java.en.When;
6   import java.io.BufferedReader;
7   import java.io.InputStreamReader;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.nio.file.Paths;
11  import static org.junit.jupiter.api.Assertions.assertTrue;
12  import static org.junit.jupiter.api.Assertions.fail;
13  
14  public class CodeQualityStepDefinitions {
15  
16      private int exitCode = -1;
17      private StringBuilder output = new StringBuilder();
18  
19      @Given("the backend source code is available")
20      public void backend_source_code_available() {
21          Path backendPath = Paths.get("src/main/java");
22          assertTrue(Files.exists(backendPath), "Backend source directory not found");
23      }
24  
25      @When("I run the compilation with the 'development' profile")
26      public void run_compilation_with_development_profile() throws Exception {
27          // We use the maven wrapper to run pmd:check which is part of the development profile
28          ProcessBuilder builder = new ProcessBuilder("./mvnw", "pmd:check", "-Pdevelopment", "-pl", "backend");
29          builder.directory(Paths.get("..").toAbsolutePath().toFile());
30          builder.redirectErrorStream(true);
31          
32          Process process = builder.start();
33          try (BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream()))) {
34              String line;
35              while ((line = reader.readLine()) != null) {
36                  output.append(line).append("\n");
37              }
38          }
39          exitCode = process.waitFor();
40      }
41  
42      @Then("the build should fail if there are unused imports or redundant annotations")
43      public void build_should_fail_if_violations() {
44          if (exitCode != 0) {
45              fail("Code quality check failed with exit code " + exitCode + ". Output:\n" + output.toString());
46          }
47      }
48  }
