150     @Given("there is a(n) {word} request for table {string}")
151     public void there_is_initial_status_request_for_table(String status, String table) {
152         String requestId = java.util.UUID.randomUUID().toString();
153         String finalStatus = status.toUpperCase();
154         if ("PENDING".equals(finalStatus) || "APPROVED".equals(finalStatus)) {
155             givenAuth()
156                 .contentType(ContentType.JSON)
157                 .body(List.of(Map.of(
158                     "id", requestId,
159                     "catalogName", "uc-oss",
160                     "schemaName", "default",
161                     "tableName", table,
162                     "privileges", List.of("SELECT"),
163                     "status", finalStatus,
164                     "justification", "Initial state test"
165                 )))
166                 .post("/api/storage/requests")
167                 .then().statusCode(200);
168 
169             if ("APPROVED".equals(finalStatus)) {
170                 String path = "/" + "uc-oss" + "/" + "default" + "/" + table;
171                 // Use a default user for now
172                 catalogService.applyPolicy("uc-oss", path, "SELECT", "admin");
173             }
174         } else {
175             throw new IllegalArgumentException("Unknown status: " + status);
176         }
177     }

1128     @When("I connect to the audit SSE stream at {string}")
1129     public void connect_to_sse_stream(String path) {
1130         sseEvents.clear();
1131         String url = RestAssured.baseURI + ":" + RestAssured.port + path;
1132         LOG.info("Connecting to SSE stream at: " + url);
1133         java.util.concurrent.CompletableFuture.runAsync(() -> {
1134             try {
1135                 java.net.http.HttpClient client = java.net.http.HttpClient.newHttpClient();
1136                 java.net.http.HttpRequest request = java.net.http.HttpRequest.newBuilder()
1137                     .uri(java.net.URI.create(url))
1138                     .header("Authorization", "Bearer " + currentToken)
1139                     .header("Accept", "text/event-stream")
1140                     .build();
1141                 
1142                 client.sendAsync(request, java.net.http.HttpResponse.BodyHandlers.ofLines())
1143                     .thenAccept(res -> {
1144                         res.body().forEach(line -> {
1145                             if (line.startsWith("data:")) {
1146                                 String data = line.substring(5).trim();
1147                                  try {
1148                                     Map<String, Object> event = new com.fasterxml.jackson.databind.ObjectMapper().readValue(data, new com.fasterxml.jackson.core.type.TypeReference<Map<String, Object>>() {});
1149                                     sseEvents.add(event);
1150                                 } catch (Exception e) {}
1151                             }
1152                         });
1153                     }).join();
1154             } catch (Exception e) {
1155                 LOG.error("SSE Connection error", e);
1156             }
1157         });
1158         // Give it a moment to connect
1159         try { Thread.sleep(500); } catch (InterruptedException e) {}
1160     }
