1   package com.rhlowery.acs;
2   
3   import io.cucumber.java.en.Given;
4   import io.cucumber.java.en.Then;
5   import io.cucumber.java.en.When;
6   import io.restassured.RestAssured;
7   import io.restassured.http.ContentType;
8   import io.restassured.response.Response;
9   import java.util.Map;
10  import static org.hamcrest.Matchers.*;
11  
12  public class CatalogStepDefinitions {
13  
14      private Response lastResponse;
15      private String lastPrincipal;
16  
17      private io.restassured.specification.RequestSpecification givenAuth() {
18          return RestAssured.given()
19              .contentType(ContentType.JSON);
20      }
21  
22      @When("I request the children of path {string} from catalog {string}")
23      public void i_request_the_children_of_path_from_catalog(String path, String catalogId) {
24          lastResponse = givenAuth()
25              .queryParam("path", path)
26              .get("/api/catalog/" + catalogId + "/nodes");
27      }
28  
29      @Then("the response should contain the node {string}")
30      public void the_response_should_contain_the_node(String name) {
31          lastResponse.then().body("name", hasItem(name));
32      }
33  
34      @Then("the node type should be {string}")
35      public void the_node_type_should_be(String type) {
36          lastResponse.then().body("type", hasItem(type));
37      }
38  
39      @Then("the implementation should be {string}")
40      public void the_implementation_should_be(String impl) {
41          lastResponse.then().body("implementation", hasItem(impl));
42      }
43  
44      @Given("there is a catalog node at path {string} in catalog {string}")
45      public void there_is_a_catalog_node_at_path_in_catalog(String path, String catalogId) {
46          givenAuth()
47              .queryParam("path", path)
48              .get("/api/catalog/" + catalogId + "/nodes/verify")
49              .then().statusCode(200);
50      }
51  
52      @When("I {word} access for principal {string} on node {string} in catalog {string}")
53      public void i_action_access_for_principal_on_node_in_catalog(String action, String principal, String path, String catalogId) {
54          this.lastPrincipal = principal;
55          lastResponse = givenAuth()
56              .body(Map.of(
57                  "action", action,
58                  "principal", principal,
59                  "path", path
60              ))
61              .post("/api/catalog/" + catalogId + "/nodes/policy");
62      }
63  
64      @Then("the audit log should record the {string} event")
65      public void the_audit_log_should_record_the_event(String event) {
66          // Check audit log
67          givenAuth()
68              .get("/api/audit/log")
69              .then()
70              .statusCode(200);
71      }
72  
73      @Then("the node {string} in catalog {string} should have effective permissions {string}")
74      public void the_node_in_catalog_should_have_effective_permissions(String path, String catalogId, String permissions) {
75          String principal = lastPrincipal != null ? lastPrincipal : "alice"; 
76          if (principal.equals("alice") && path.contains("hr")) principal = "bob";
77          if (principal.equals("alice") && path.contains("sys")) principal = "charlie";
78  
79          givenAuth()
80              .queryParam("path", path)
81              .queryParam("principal", principal)
82              .get("/api/catalog/" + catalogId + "/nodes/permissions")
83              .then()
84              .statusCode(200)
85              .body("effective", equalTo(permissions));
86      }
87  
88      @When("I list registered catalog providers")
89      public void i_list_registered_catalog_providers() {
90          lastResponse = givenAuth().get("/api/catalog/providers");
91      }
92  
93      @Then("I verify that I see in the provider list {string}")
94      public void i_verify_that_I_see_in_the_provider_list(String text) {
95          lastResponse.then().body(containsString(text));
96      }
97  }
