9   import jakarta.enterprise.context.ApplicationScoped;
10  import jakarta.inject.Inject;
11  import io.smallrye.mutiny.Uni;
12  import java.util.HashSet;
13  import java.util.List;

17  /**
18   * Augments the authenticated SecurityIdentity with ACS-specific personas and groups.
19   * This ensures that whether the user logs in via the BFF (JWT) or directly via OIDC,
20   * the persona mapping logic remains consistent.
21   *
22   * [PlantUML: Security identity augmentation workflow with persona resolution]
23   */
24  @ApplicationScoped
25  public class AcsSecurityIdentityAugmentor implements SecurityIdentityAugmentor {
26  
27    @Inject
28    UserService userService;
29  
30    /**
31     * Augments the security identity with additional information from the ACS database and JWT claims.
32     *
33     * @param identity The initial security identity
34     * @param context The authentication request context
35     * @return A Uni providing the augmented security identity
36     */
37    @Override
38    public Uni<SecurityIdentity> augment(SecurityIdentity identity, AuthenticationRequestContext context) {
39      if (identity.isAnonymous()) {
40        return Uni.createFrom().item(identity);
41      }
42  
43      return Uni.createFrom().item(identity).onItem().transform(id -> {
44        String userId = id.getPrincipal().getName();
45        Set<String> roles = id.getRoles();
46  
47        // 1. Resolve Persona (Database -> Group -> Token -> Default)
48        String persona = null;
49  
50        // Priority 1: User-specific persona in local database
51        Optional<User> localUser = userService.getUser(userId);
52        if (localUser.isPresent() && localUser.get().persona() != null) {
53          persona = localUser.get().persona();
54        }
55  
56        // Priority 2: Group-based persona resolution
57        if (persona == null && !roles.isEmpty()) {
58          for (String groupId : roles) {
59            Optional<com.rhlowery.acs.domain.Group> g = userService.getGroup(groupId);
60            if (g.isPresent() && g.get().persona() != null) {
61              persona = g.get().persona();
62              break;
63            }
64          }
65        }
66  
67        // Priority 3: Persona defined in JWT token (OIDC provider)
68        if (persona == null && identity.getPrincipal() instanceof org.eclipse.microprofile.jwt.JsonWebToken) {
69          org.eclipse.microprofile.jwt.JsonWebToken jwt = (org.eclipse.microprofile.jwt.JsonWebToken) identity.getPrincipal();
70          persona = jwt.getClaim("persona");
71        }
72  
73        // 2. Provision user if not exists (JIT)
74        if (localUser.isEmpty()) {
75          User newUser = new User(userId, userId, userId + "@example.com", "STANDARD_USER", List.copyOf(roles), persona);
76          userService.saveUser(newUser);
77        }
78  
79        // 3. Build augmented identity
80        QuarkusSecurityIdentity.Builder builder = QuarkusSecurityIdentity.builder(id);
81  
82        // Collect all roles to add
83        Set<String> rolesToApply = new HashSet<>(roles);
84  
85        if (localUser.isPresent() && localUser.get().groups() != null) {
86          rolesToApply.addAll(localUser.get().groups());
87        }
88  
89        if (persona != null) {
90          rolesToApply.add(persona);
91          builder.addAttribute("persona", persona);
92        }
93  
94        // 4. Ensure admin personas get corresponding roles, unless explicitly restricted to REQUESTER
95        if (!"REQUESTER".equals(persona) && (rolesToApply.contains("admins") || "ADMIN".equals(persona) || "GOVERNANCE_ADMIN".equals(persona) || "SECURITY_ADMIN".equals(persona))) {
96          rolesToApply.add("ADMIN");
97          rolesToApply.add("SECURITY_ADMIN");
98          rolesToApply.add("AUDITOR");
99        }
100 
101       for (String role : rolesToApply) {
102         builder.addRole(role);
103       }
104 
105       return (SecurityIdentity) builder.build();
106     }).runSubscriptionOn(io.smallrye.mutiny.infrastructure.Infrastructure.getDefaultWorkerPool());
107   }
108 }
