12  /**
13   * Base class for mock {@link CatalogProvider} implementations.
14   * Provides default logic for H2-backed policy storage and static metadata
15   * discovery for various simulated catalog types (Polaris, Atlan, etc.).
16   *
17   * [PlantUML: Inheritance hierarchy for mock providers showing policy storage and discovery]
18   */
19  public abstract class AbstractMockProvider implements CatalogProvider {
20    protected final String catalogId;
21    protected final String implClass;
22  
23    protected final Map<String, String> policies = new ConcurrentHashMap<>();
24  
25    protected AbstractMockProvider(String catalogId, String implClass) {
26      this.catalogId = catalogId;
27      this.implClass = implClass;
28    }
29  
30    public AbstractMockProvider(String catalogId) {
31      this(catalogId, "MockProvider");
32    }
33  
34    @Override
35    public String getProviderName() {
36      return catalogId.substring(0, 1).toUpperCase() + catalogId.substring(1) + " Provider";
37    }
38  
39    @Override
40    public Map<String, String> getCapabilities() {
41      Map<String, String> caps = new HashMap<>();
42      caps.put("type", "mock");
43      caps.put("discovery", "static");
44      return caps;
45    }
46  
47    @Override
48    public String getCatalogId() {
49      return catalogId;
50    }
51  
52    @Override
53    public List<CatalogNode> getChildren(String path) {
54      List<CatalogNode> nodes = new ArrayList<>();
55      if (path == null || "/".equals(path)) {
56        if ("polaris".equals(catalogId)) {
57          nodes.add(new CatalogNode("polaris_cat", NodeType.CATALOG, "/polaris_cat", implClass, List.of("polaris-admins"), "admin", "Polaris catalog", "FULL_ACCESS"));
58        } else if ("datahub".equals(catalogId)) {
59          nodes.add(new CatalogNode("default", NodeType.DATABASE, "/default", implClass, List.of("datahub-admins"), "admin", "DataHub database", "FULL_ACCESS"));
60        } else if ("gravitino".equals(catalogId)) {
61          nodes.add(new CatalogNode("metalake", NodeType.NAMESPACE, "/metalake", implClass, List.of("gravitino-admins"), "admin", "Gravitino metalake", "FULL_ACCESS"));
62        } else if ("atlan".equals(catalogId)) {
63          nodes.add(new CatalogNode("asset", NodeType.TABLE, "/asset", implClass, List.of("atlan-admins"), "admin", "Atlan asset", "FULL_ACCESS"));
64        } else if ("hive".equals(catalogId)) {
65          nodes.add(new CatalogNode("hms_cat", NodeType.CATALOG, "/hms_cat", implClass, List.of("hms-admins"), "admin", "Hive Metastore", "FULL_ACCESS"));
66        } else {
67          nodes.add(new CatalogNode("main", NodeType.CATALOG, "/main", implClass, List.of("admins"), "admin", "Main catalog", "FULL_ACCESS"));
68          nodes.add(new CatalogNode("default", NodeType.DATABASE, "/default", implClass, List.of("admins"), "admin", "Default database", "FULL_ACCESS"));
69        }
70      } else if ("/main".equals(path)) {
71        nodes.add(new CatalogNode("default", NodeType.DATABASE, "/main/default", implClass, List.of("admins"), "admin", "Default database in main", "FULL_ACCESS"));
72      } else if (path.contains("default")) {
73        nodes.add(new CatalogNode("sensitive_tbl", NodeType.TABLE, path + "/sensitive_tbl", implClass, List.of("sensitive-approvers"), "data-governor", "Highly sensitive data", "READ_ONLY"));
74      } else if (path.contains("finance")) {
75        nodes.add(new CatalogNode("quarters", NodeType.NAMESPACE, path + "/quarters", implClass, List.of("finance-approvers"), "finance-lead", "Financial reports", "NO_ACCESS"));
76      }
77      return nodes;
78    }
79  
80    @Override
81    public CatalogNode getNode(String path) {
82      if (path == null || "/".equals(path)) return new CatalogNode("root", NodeType.NAMESPACE, "/", implClass, List.of("admins"), "admin", "Root node", "ADMIN");
83      if (path.contains("sensitive")) return new CatalogNode("sensitive_tbl", NodeType.TABLE, path, implClass, List.of("sensitive-approvers"), "data-governor", "Sensitive table", "READ");
84      if (path.contains("salaries")) return new CatalogNode("salaries", NodeType.TABLE, path, implClass, List.of("finance-leads"), "finance_lead", "Salary info", "NONE");
85      if (path.contains("finance")) return new CatalogNode("finance", NodeType.NAMESPACE, path, implClass, List.of("finance-approvers"), "finance_lead", "Finance dept", "READ");
86      if (path.contains("staged")) return new CatalogNode("data_uploads", NodeType.VOLUME, path, implClass, List.of("data-governors"), "admin", "Staging area", "FULL");
87      if (path.contains("model")) return new CatalogNode("risk-model", NodeType.MODEL, path, implClass, List.of("model-owners"), "admin", "ML Risk Model", "ACCESS");
88      if (path.contains("compute")) return new CatalogNode("cluster-01", NodeType.COMPUTE, path, implClass, List.of("platform-admins"), "admin", "Databricks Cluster", "USAGE");
89      if (path.contains("default")) return new CatalogNode("default", NodeType.DATABASE, path, implClass, List.of("default-admins"), "admin", "Default DB", "OWNER");
90      return new CatalogNode("node", NodeType.TABLE, path, implClass, List.of("admins"), "admin", "Generic node", "READ");
91    }
92  
93    @Override
94    public void applyPolicy(String path, String action, String principal) {
95      if ("overlay".equalsIgnoreCase(action) || "revoke".equalsIgnoreCase(action) || "approve".equalsIgnoreCase(action)) {
96        return;
97      }
98      policies.put(principal + ":" + path, action);
99    }
100 
101   @Override
102   public String getEffectivePermissions(String path, String principal) {
103     String action = policies.get(principal + ":" + path);
104     if (action != null) return action;
105 
106     if (path != null && (path.contains("salaries") || path.contains("users") || path.contains("polaris"))) {
107       if ("alice".equals(principal)) return "SELECT";
108       if ("polaris".equals(principal)) return "READ";
109       if ("bob".equals(principal)) return "NONE";
110       if ("charlie".equals(principal)) return "READ";
111     }
112     if (path != null && path.contains("sensitive")) {
113       if ("alice".equals(principal)) return "READ";
114       if ("bob".equals(principal)) return "NONE";
115       if ("charlie".equals(principal)) return "READ";
116     }
117     return "READ";
118   }
119 
120   /**
121    * Clears all simulated policies.
122    */
123   public void clear() {
124     policies.clear();
125   }
