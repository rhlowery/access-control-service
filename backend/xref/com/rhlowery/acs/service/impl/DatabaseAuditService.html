4   import com.rhlowery.acs.infrastructure.entity.AuditEntryEntity;
5   import com.rhlowery.acs.service.AuditService;
6   import com.fasterxml.jackson.core.type.TypeReference;
7   import com.fasterxml.jackson.databind.ObjectMapper;
8   import io.smallrye.mutiny.Multi;
9   import io.smallrye.mutiny.operators.multi.processors.BroadcastProcessor;
10  import jakarta.enterprise.context.ApplicationScoped;
11  import java.util.List;
12  import java.util.Map;
13  import java.util.stream.Collectors;
14  import org.jboss.logging.Logger;
15  
16  /**
17   * JPA-based implementation of the {@link AuditService}.
18   * Handles persistent storage of audit entries and real-time broadcasting
19   * of events using Mutiny {@link io.smallrye.mutiny.Multi}.
20   */
21  @ApplicationScoped
22  public class DatabaseAuditService implements AuditService {
23    private static final Logger LOG = Logger.getLogger(DatabaseAuditService.class);
24    private final ObjectMapper mapper = new ObjectMapper();
25    private final BroadcastProcessor<AuditEntry> processor = BroadcastProcessor.create();
26  
27    /**
28     * Persists a new audit entry and broadcasts it to any active subscribers.
29     *
30     * @param entry The audit entry to record
31     */
32    @Override
33    @jakarta.transaction.Transactional
34    public void log(AuditEntry entry) {
35      LOG.debug("Logging audit entry: " + entry.type() + " by " + entry.actor());
36      String detailsStr = "";
37      try {
38        detailsStr = mapper.writeValueAsString(entry.details());
39      } catch (Exception e) {
40        LOG.error("Failed to serialize audit details", e);
41      }
42  
43      AuditEntryEntity entity = new AuditEntryEntity(
44        entry.id(), entry.type(), entry.actor(), entry.userId(), 
45        entry.timestamp(), entry.serverTimestamp(), detailsStr, 
46        entry.signature(), entry.signer()
47      );
48      entity.persist();
49      processor.onNext(entry); // Broadcast to active streams
50    }
51  
52    /**
53     * Retrieves all historical audit logs from the database.
54     *
55     * @return A list of domain {@link AuditEntry} objects
56     */
57    @Override
58    public List<AuditEntry> getLogs() {
59      return AuditEntryEntity.<AuditEntryEntity>listAll().stream()
60        .map(this::mapToDomain)
61        .collect(Collectors.toList());
62    }
63  
64    /**
65     * Provides a live stream of audit events as they are recorded.
66     *
67     * @return A Multi stream of {@link AuditEntry} objects
68     */
69    @Override
70    public Multi<AuditEntry> streamLogs() {
71      return processor;
72    }
73  
74      private AuditEntry mapToDomain(AuditEntryEntity entity) {
75          Map<String, Object> details = Map.of();
76          try {
77              details = mapper.readValue(entity.details, new TypeReference<Map<String, Object>>() {});
78          } catch (Exception e) {
79              LOG.error("Failed to deserialize audit details for entry " + entity.id, e);
80          }
81          
82          return new AuditEntry(
83              entity.id, entity.type, entity.actor, entity.userId, 
84              entity.timestamp, entity.serverTimestamp, details, 
85              entity.signature, entity.signer
86          );
87      }
88  }
