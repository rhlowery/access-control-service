11  /**
12   * Mock Identity Provider for local development and demo purposes.
13   * Ships with a default platform administrator (admin/admin).
14   * Additional users can register themselves via the register() method.
15   */
16  @ApplicationScoped
17  public class MockIdentityProvider implements IdentityProvider {
18  
19      private static final Logger LOG = Logger.getLogger(MockIdentityProvider.class);
20  
21      /** userId -> MockUser record */
22      private final Map<String, MockUser> users = new ConcurrentHashMap<>();
23  
24      record MockUser(String userId, String password, String name, String email, String persona, List<String> groups) {}
25  
26      public MockIdentityProvider() {
27          // Seed default platform administrator
28          users.put("admin", new MockUser(
29              "admin", "admin", "Platform Administrator", "admin@localhost",
30              "PLATFORM_ADMIN", List.of("admins", "governance-team")
31          ));
32      }
33  
34      @Override
35      public String getId() {
36          return "mock";
37      }
38  
39      @Override
40      public String getName() {
41          return "Mock Identity Provider";
42      }
43  
44      @Override
45      public String getType() {
46          return "MOCK";
47      }

49      @Override
50      public Optional<Map<String, Object>> authenticate(Map<String, Object> credentials) {
51          String userId = (String) credentials.get("userId");
52          String password = (String) credentials.get("password");
53  
54          if (userId == null || userId.isBlank()) {
55              return Optional.empty();
56          }
57  
58          MockUser user = users.get(userId);
59          if (user != null) {
60              // Existing user — validate password
61              if (password == null || !password.equals(user.password())) {
62                  LOG.warn("Mock IdP: invalid password for user '" + userId + "'");
63                  return Optional.empty();
64              }
65              
66              // Update groups on every login to reflect the current test scenario
67              @SuppressWarnings("unchecked")
68              List<String> updatedGroups = (List<String>) credentials.get("groups");
69              if (updatedGroups != null) {
70                  user = new MockUser(user.userId(), user.password(), user.name(), user.email(), (String)credentials.get("persona"), updatedGroups);
71                  users.put(userId, user);
72                  LOG.infof("Mock IdP: authenticated existing user '%s' with updated groups %s", userId, updatedGroups);
73              }
74              
75              Map<String, Object> result = new java.util.HashMap<>();
76              result.put("userId", user.userId());
77              result.put("userName", user.name());
78              result.put("email", user.email());
79              result.put("providerId", getId());
80              result.put("groups", user.groups());
81              if (user.persona() != null) {
82                  result.put("persona", user.persona());
83              }
84              return Optional.of(result);
85          }
86  
87          // Unknown user — auto-register with the supplied password, groups, and persona (dev/test convenience)
88          if (password != null && !password.isBlank()) {
89              @SuppressWarnings("unchecked")
90              List<String> groups = (List<String>) credentials.get("groups");
91              if (groups == null) groups = List.of();
92              
93              String persona = (String) credentials.get("persona");
94  
95              MockUser newUser = new MockUser(userId, password, userId, userId + "@mock.local", persona, groups);
96              users.put(userId, newUser);
97              LOG.infof("Mock IdP: auto-registered new user '%s' with groups %s and persona %s", userId, groups, persona);
98              Map<String, Object> result = new java.util.HashMap<>();
99              result.put("userId", userId);
100             result.put("userName", userId);
101             result.put("email", newUser.email());
102             result.put("providerId", getId());
103             result.put("groups", groups);
104             if (persona != null) {
105                 result.put("persona", persona);
106             }
107             return Optional.of(result);
108         }
109 
110         LOG.warn("Mock IdP: unknown user '" + userId + "' and no password provided");
111         return Optional.empty();
112     }
113 
114     @Override
115     public List<String> getGroups(String userId) {
116         MockUser user = users.get(userId);
117         return user != null ? user.groups() : List.of();
118     }
119 
120     /**
121      * Programmatic registration for tests or runtime use.
122      */
123     public void register(String userId, String password, String name, String persona, List<String> groups) {
124         users.put(userId, new MockUser(userId, password, name, userId + "@mock.local", persona, groups));
125     }
126 
127     /**
128      * Check if a given userId exists in the mock store.
129      */
130     public boolean hasUser(String userId) {
131         return users.containsKey(userId);
132     }
