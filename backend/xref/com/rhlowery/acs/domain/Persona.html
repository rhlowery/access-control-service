3   import io.quarkus.runtime.annotations.RegisterForReflection;
4   
5   /**
6    * Defines the available system-wide personas in the ACS.
7    * Personas are used to drive UI capabilities and security logic.
8    *
9    * [Persona system-wide definition and assignment hierarchy]
10   */
11  @RegisterForReflection
12  public record Persona(
13    String id,
14    String name,
15    String description
16  ) {
17    public static final Persona ADMIN = new Persona("ADMIN", "Admin", "Full system access and configuration");
18    public static final Persona APPROVER = new Persona("APPROVER", "Approver", "Can review and approve/reject requests");
19    public static final Persona REQUESTER = new Persona("REQUESTER", "Requester", "Can submit and view own access requests");
20    public static final Persona SECURITY_ADMIN = new Persona("SECURITY_ADMIN", "Security Admin", "System-wide security configuration and policy management");
21    public static final Persona PLATFORM_ADMIN = new Persona("PLATFORM_ADMIN", "Platform Admin", "Infrastructure and platform level administration");
22    public static final Persona GOVERNANCE_ADMIN = new Persona("GOVERNANCE_ADMIN", "Governance Admin", "Default data governance and catalog level administration");
23    public static final Persona AUDITOR = new Persona("AUDITOR", "Auditor", "Can view and stream full audit logs for historical or real-time review");
24    public static final Persona REVIEWER = new Persona("REVIEWER", "Reviewer", "Has read-only access to audit dashboards and live streams");
25  
26    /**
27     * Returns a list of all predefined system personas.
28     *
29     * @return List of Persona objects
30     */
31    public static java.util.List<Persona> all() {
32      return java.util.List.of(ADMIN, APPROVER, REQUESTER, SECURITY_ADMIN, PLATFORM_ADMIN, GOVERNANCE_ADMIN, AUDITOR, REVIEWER);
33    }
34  }
