1   package com.rhlowery.acs.resource;
2   
3   import jakarta.ws.rs.*;
4   import jakarta.ws.rs.core.MediaType;
5   import jakarta.ws.rs.core.Response;
6   import java.util.Map;
7   import org.eclipse.microprofile.openapi.annotations.Operation;
8   import org.eclipse.microprofile.openapi.annotations.tags.Tag;
9   import org.jboss.logging.Logger;
10  
11  @jakarta.enterprise.context.ApplicationScoped
12  @Path("/api")
13  @Produces(MediaType.APPLICATION_JSON)
14  @Tag(name = "Proxy", description = "Discovery and SQL proxy endpoints for Databricks and Unity Catalog")
15  public class ProxyResource {
16  
17      private static final Logger LOG = Logger.getLogger(ProxyResource.class);
18  
19      @jakarta.inject.Inject
20      @org.eclipse.microprofile.rest.client.inject.RestClient
21      com.rhlowery.acs.infrastructure.DatabricksClient databricksClient;
22  
23      @GET
24      @Path("/sdk/{target}")
25      @Operation(summary = "Proxy SDK fetch", description = "Proxies a fetch request to the Databricks Unity Catalog SDK placeholder")
26      public Response sdkFetch(@PathParam("target") String target, 
27                              @HeaderParam("x-workspace-host") String host,
28                              @HeaderParam("Authorization") String auth) {
29          LOG.infof("SDK Fetch: target=%s host=%s", target, host);
30          try {
31              Map<String, Object> response = databricksClient.fetchFromUC(target, auth, 10, null, null, null);
32              return Response.ok(response).build();
33          } catch (Exception e) {
34              LOG.error("SDK Fetch error", e);
35              return Response.status(500).entity(Map.of("error", e.getMessage())).build();
36          }
37      }
38  
39      @GET
40      @Path("/uc/{path: .*}")
41      @Operation(summary = "Proxy Unity Catalog request", description = "Proxies a generic request to the Unity Catalog API placeholder")
42      public Response ucProxy(@PathParam("path") String path) {
43          LOG.infof("UC Proxy: path=%s", path);
44          return Response.status(501).entity(Map.of("error", "Not implemented", "path", path)).build();
45      }
46  
47      @POST
48      @Path("/sql/execute")
49      @Operation(summary = "Execute SQL", description = "Proxies a SQL execution request to the Databricks SQL Warehouse placeholder")
50      @Consumes(MediaType.APPLICATION_JSON)
51      public Response executeSql(com.rhlowery.acs.domain.SqlRequest request, @HeaderParam("Authorization") String auth) {
52          if (request == null || request.statement() == null || request.statement().trim().isEmpty()) {
53              return Response.status(400).entity(Map.of("error", "Empty statement")).build();
54          }
55          LOG.infof("SQL Execute: %s", request.statement());
56          try {
57              Map<String, Object> response = databricksClient.executeSql(auth, request);
58              return Response.ok(response).build();
59          } catch (Exception e) {
60              LOG.error("SQL Execute error", e);
61              return Response.status(500).entity(Map.of("error", e.getMessage())).build();
62          }
63      }
64  }
