1   package com.rhlowery.acs.resource;
2   
3   import com.rhlowery.acs.dto.CatalogRegistration;
4   import jakarta.ws.rs.*;
5   import jakarta.ws.rs.core.MediaType;
6   import jakarta.ws.rs.core.Response;
7   import java.util.ArrayList;
8   import java.util.Map;
9   import java.util.concurrent.ConcurrentHashMap;
10  import org.eclipse.microprofile.openapi.annotations.Operation;
11  import org.eclipse.microprofile.openapi.annotations.responses.APIResponse;
12  import org.eclipse.microprofile.openapi.annotations.tags.Tag;
13  import org.jboss.logging.Logger;
14  
15  /**
16   * REST Resource for dynamic catalog registration and management.
17   */
18  @jakarta.enterprise.context.ApplicationScoped
19  @Path("/api/catalog/registrations")
20  @Produces(MediaType.APPLICATION_JSON)
21  @Consumes(MediaType.APPLICATION_JSON)
22  @Tag(name = "Catalog Registration", description = "Endpoints for managing dynamic catalog connections")
23  public class CatalogRegistrationResource {
24  
25      private static final Logger LOG = Logger.getLogger(CatalogRegistrationResource.class);
26  
27      @jakarta.inject.Inject
28      com.rhlowery.acs.service.CatalogService catalogService;
29  
30      private final Map<String, CatalogRegistration> registrations = new ConcurrentHashMap<>();
31  
32      @POST
33      @Operation(summary = "Register a new catalog", description = "Adds a new catalog connection to the system")
34      @APIResponse(responseCode = "201", description = "Catalog registered successfully")
35      public Response registerCatalog(CatalogRegistration registration) {
36          if (registration == null || registration.id == null) {
37              return Response.status(400).entity(Map.of("error", "Missing id")).build();
38          }
39          LOG.infof("Registering catalog: %s", registration.id);
40          registrations.put(registration.id, registration);
41          
42          // If it's a mock, we can dynamically register a provider for it
43          if ("mock".equalsIgnoreCase(registration.type)) {
44              // This is a simplified example of dynamic provider instantiation
45              // In a real system, this would use a factory
46              com.rhlowery.acs.service.CatalogProvider provider = 
47                  new com.rhlowery.acs.service.impl.AbstractMockProvider(registration.id) {};
48              catalogService.registerProvider(provider);
49          }
50          
51          return Response.status(201).entity(registration).build();
52      }
53  
54      @GET
55      @Operation(summary = "List registered catalogs", description = "Returns a list of all currently registered catalog connections")
56      public Response listRegistrations() {
57          return Response.ok(new ArrayList<>(registrations.values())).build();
58      }
59  
60      @GET
61      @Path("/{id}")
62      @Operation(summary = "Get catalog details", description = "Returns details for a specific catalog registration")
63      public Response getRegistration(@PathParam("id") String id) {
64          CatalogRegistration registration = registrations.get(id);
65          if (registration == null) return Response.status(404).build();
66          return Response.ok(registration).build();
67      }
68  
69      @PATCH
70      @Path("/{id}")
71      @Operation(summary = "Update catalog settings", description = "Partially updates the settings for an existing catalog registration")
72      public Response updateRegistration(@PathParam("id") String id, CatalogRegistration update) {
73          CatalogRegistration existing = registrations.get(id);
74          if (existing == null) return Response.status(404).build();
75          
76          LOG.infof("Updating catalog: %s", id);
77          
78          if (update.name != null) existing.name = update.name;
79          if (update.type != null) existing.type = update.type;
80          if (update.settings != null) {
81              if (existing.settings == null) {
82                  existing.settings = new java.util.HashMap<>();
83              }
84              existing.settings.putAll(update.settings);
85          }
86          
87          return Response.ok(existing).build();
88      }
89  
90      @DELETE
91      @Path("/{id}")
92      @Operation(summary = "Unregister a catalog", description = "Removes a catalog connection from the system")
93      @APIResponse(responseCode = "204", description = "Deleted")
94      public Response deleteRegistration(@PathParam("id") String id) {
95          if (registrations.remove(id) == null) return Response.status(404).build();
96          return Response.noContent().build();
97      }
98  }
