32   * REST Resource for managing the lifecycle of data access requests.
33   * Supports submission, approval, rejection, and verification of requests.
34   * Provides real-time updates via Server-Sent Events (SSE).
35   *
36   * <pre>
37   * {@code
38   * @startuml
39   * actor User
40   * participant "AccessRequestResource" as API
41   * participant "AccessRequestService" as Service
42   * participant "CatalogService" as Catalog
43   *
44   * User -> API : Submit Request
45   * API -> Catalog : getRequiredApprovers()
46   * Catalog --> API : Approver Groups
47   * API -> Service : saveRequest(PENDING)
48   * Service --> API : Saved Request
49   * API -> User : 202 Accepted
50   * @enduml
51   * }
52   * </pre>

103     public HalAccessRequest(AccessRequest r) {
104       this.id = r.id();
105       this.catalogName = r.catalogName();
106       this.schemaName = r.schemaName();
107       this.tableName = r.tableName();
108       this.privileges = r.privileges();
109       this.status = r.status();
110       this.justification = r.justification();
111       this.rejectionReason = r.rejectionReason();
112       this.requesterId = r.requesterId();
113       this.userId = r.userId();
114       this.principalType = r.principalType();
115       this.resourceType = r.resourceType();
116       this.approverGroups = r.approverGroups();
117       this.expirationTime = r.expirationTime();
118       Map<String, Object> links = new java.util.HashMap<>();
119       links.put("self", Map.of("href", "/api/storage/requests/" + r.id()));
120       if ("PENDING".equals(r.status()) || "PARTIALLY_APPROVED".equals(r.status())) {
121         links.put("approve", Map.of("href", "/api/storage/requests/" + r.id() + "/approve"));
122         links.put("reject", Map.of("href", "/api/storage/requests/" + r.id() + "/reject"));
123       } else if ("APPROVED".equals(r.status())) {
124         links.put("verify", Map.of("href", "/api/storage/requests/" + r.id() + "/verify"));
125       }
126       this._links = links;
127     }
128   }

130   /**
131    * Establishes a Server-Sent Events (SSE) connection for real-time status updates.
132    *
133    * @param eventSink The SSE sink provided by the JAX-RS runtime
134    * @return A Response 200 indicating connection established
135    */
136   @GET
137   @Path("/stream")
138   @Produces(MediaType.SERVER_SENT_EVENTS)
139   @Operation(summary = "Stream requests", description = "Provides a Server-Sent Events stream for real-time access request updates")
140   public Response stream(@Context SseEventSink eventSink) {
141     if (sse != null) {
142       sinks.add(eventSink);
143       OutboundSseEvent event = sse.newEventBuilder()
144         .name("connected")
145         .data("SSE stream active")
146         .build();
147       eventSink.send(event);
148     }
149     return Response.ok().build();
150   }

152   /**
153    * Internal helper to broadcast events to all active SSE subscribers.
154    *
155    * @param name The event name
156    * @param data The event payload
157    */
158   private void broadcast(String name, Object data) {
159     if (sse != null) {
160       OutboundSseEvent event = sse.newEventBuilder()
161         .name(name)
162         .data(data)
163         .build();
164       sinks.removeIf(sink -> {
165         try {
166           sink.send(event);
167           return false;
168         } catch (Exception e) {
169           return true;
170         }
171       });
172     }
173   }

188 
189     LOG.infof("Listing requests for user: %s, persona=%s, isAdmin=%b", userId, persona, isAdmin);
190     List<HalAccessRequest> halRequests = accessRequestService.getAllRequests(userId, groups, isAdmin).stream()

192       .collect(Collectors.toList());
193 
194     return Response.ok(halRequests).build();

213   /**
214    * Submits one or more new access requests. Performs auto-resolution of required approvers.
215    *
216    * @param requests List of requests submitted by the client
217    * @param securityContext JAX-RS security context
218    * @return Response containing submission status
219    */
220   @POST
221   @Operation(summary = "Submit access requests", description = "Submits one or more access requests for review")
222   public Response createRequests(List<AccessRequest> requests, @Context SecurityContext securityContext) {
223     if (requests == null || requests.isEmpty()) {
224       return Response.status(400).entity(Map.of("error", "Request list cannot be empty")).build();
225     }
226 
227     String userId = securityIdentity.getPrincipal() != null ? securityIdentity.getPrincipal().getName() : "anonymous";
228     List<String> groups = new ArrayList<>(securityIdentity.getRoles());
229     String persona = securityIdentity.getAttribute("persona");
230     boolean isAdmin = !"REQUESTER".equals(persona) && (securityIdentity.hasRole("ADMIN") || securityIdentity.hasRole("SECURITY_ADMIN") || securityIdentity.hasRole("APPROVER") || securityIdentity.hasRole("GOVERNANCE_ADMIN") || securityIdentity.hasRole("AUDITOR") || groups.contains("admins"));
231 
232     for (AccessRequest r : requests) {
233       String path = "/" + r.catalogName() + "/" + r.schemaName() + "/" + r.tableName();
234       List<String> requiredApprovers = new ArrayList<>();
235 
236       // 1. Resolve approvers from catalog
237       List<String> catalogApprovers = catalogService.getRequiredApprovers(r.catalogName(), path);
238       if (catalogApprovers != null) {
239         requiredApprovers.addAll(catalogApprovers);
240       }
241 
242       // 2. Add governance-team if not present (Mandatory Governance)
243       if (!requiredApprovers.contains("governance-team")) {
244         requiredApprovers.add("governance-team");
245       }
246 
247       AccessRequest enriched = new AccessRequest(
248         r.id(),
249         userId,
250         r.userId() != null ? r.userId() : userId,
251         r.principalType() != null ? r.principalType() : "USER",
252         r.catalogName(),
253         r.schemaName(),
254         r.tableName(),
255         r.resourceType() != null ? r.resourceType() : "TABLE",
256         r.privileges(),
257         isAdmin && r.status() != null ? r.status() : "PENDING",
258         System.currentTimeMillis(),
259         null,
260         r.justification(),
261         null, // rejectionReason is null for new requests
262         requiredApprovers,
263         r.metadata() != null ? r.metadata() : new java.util.HashMap<>(),
264         r.expirationTime()
265       );
266       accessRequestService.saveRequests(List.of(enriched), userId, groups, isAdmin);
267       lineageService.emitAccessRequestEvent(enriched, userId);
268     }
269 
270     broadcast("request-created", requests);
271     return Response.ok(Map.of("status", "success", "count", requests.size())).build();
272   }

274   /**
275    * Approves a request. Performs multi-stage approval tracking if multiple groups are required.
276    * If fully approved, orchestrates the actual policy application to the target catalog.
277    *
278    * <pre>
279    * {@code
280    * @startuml
281    * start
282    * :Admin/Approver clicks Approve;
283    * :Verify authorization;
284    * :Update approvals metadata;
285    * if (All required groups approved OR current user is Admin?) then (yes)
286    *   :Set status to APPROVED;
287    *   :Grant access in Target Catalog;
288    * else (no)
289    *   :Set status to PARTIALLY_APPROVED;
290    * endif
291    * :Persist and Broadcast;
292    * stop
293    * @enduml
294    * }
295    * </pre>
296    *
297    * @param id The request ID
298    * @param securityContext JAX-RS security context
299    * @return Response containing the updated HalAccessRequest
300    */
301   @POST
302   @Path("/{id}/approve")
303   @Operation(summary = "Approve access request", description = "Approves a pending access request (Admins only)")
304   public Response approveRequest(@PathParam("id") String id, @Context SecurityContext securityContext) {
305     String userId = securityIdentity.getPrincipal() != null ? securityIdentity.getPrincipal().getName() : "anonymous";
306     List<String> groups = new ArrayList<>(securityIdentity.getRoles());
307     String persona = securityIdentity.getAttribute("persona");
308     boolean isAdmin = !"REQUESTER".equals(persona) && (securityIdentity.hasRole("ADMIN") || securityIdentity.hasRole("SECURITY_ADMIN") || securityIdentity.hasRole("APPROVER") || securityIdentity.hasRole("GOVERNANCE_ADMIN") || securityIdentity.hasRole("AUDITOR") || groups.contains("admins"));
309 
310     return accessRequestService.getRequestById(id)
311       .map(r -> {
312         if (!"PENDING".equals(r.status()) && !"PARTIALLY_APPROVED".equals(r.status())) {
313           return Response.status(400).entity(Map.of("error", "Request is not in a state that can be approved")).build();
314         }
315 
316         List<String> userGroups = new ArrayList<>(securityIdentity.getRoles());
317         boolean isDesignatedApprover = !"REQUESTER".equals(persona) && r.approverGroups() != null && r.approverGroups().stream().anyMatch(userGroups::contains);
318 
319         boolean isAuthorized = isAdmin || isDesignatedApprover;
320 
321         if (!isAuthorized) {
322           return Response.status(403).entity(Map.of("error", "You are not an authorized approver for this request")).build();
323         }
324 
325         Map<String, Object> meta = r.metadata() != null ? new java.util.HashMap<>(r.metadata()) : new java.util.HashMap<>();
326         @SuppressWarnings("unchecked")
327         List<String> signs = (List<String>) meta.getOrDefault("approvals", new ArrayList<String>());
328 
329         // Track which group this user is approving for
330         if (isDesignatedApprover && !isAdmin) {
331           String approvingGroup = r.approverGroups().stream().filter(userGroups::contains).findFirst().get();
332           if (!signs.contains(approvingGroup)) {
333             signs.add(approvingGroup);
334           }
335         }
336         meta.put("approvals", signs);
337 
338         boolean isSystemApprover = securityIdentity.hasRole("APPROVER") || securityIdentity.hasRole("GOVERNANCE_ADMIN");
339         boolean fullyApproved = isAdmin || isSystemApprover || r.approverGroups() == null || r.approverGroups().isEmpty() || signs.size() >= r.approverGroups().size();
340         String newStatus = fullyApproved ? "APPROVED" : "PARTIALLY_APPROVED";
341 
342         AccessRequest updated = new AccessRequest(
343           r.id(), r.requesterId(), r.userId(), r.principalType(), r.catalogName(), r.schemaName(), r.tableName(),
344           r.resourceType(), r.privileges(), newStatus, r.createdAt(), System.currentTimeMillis(),
345           r.justification(), r.rejectionReason(), r.approverGroups(), meta, r.expirationTime()
346         );
347 
348         if (fullyApproved) {
349           try {
350             String principal = r.userId() != null ? r.userId() : r.requesterId();
351             String path = "/" + r.catalogName() + "/" + r.schemaName() + "/" + r.tableName();
352 
353             catalogService.applyPolicy(r.catalogName(), path, r.privileges().get(0), principal);
354           } catch (Exception e) {
355             LOG.error("Failed to grant access on UC: " + e.getMessage(), e);
356           }
357         }
358 
359         accessRequestService.saveRequests(List.of(updated), userId, groups, isAdmin);
360         broadcast("request-updated", List.of(updated));
361         return Response.ok(new HalAccessRequest(updated)).build();
362       })
363       .orElse(Response.status(404).entity(Map.of("error", "Request not found")).build());
364   }

384   public Response rejectRequest(@PathParam("id") String id, RejectionRequest rejection, @Context SecurityContext securityContext) {
385     String userId = securityIdentity.getPrincipal() != null ? securityIdentity.getPrincipal().getName() : "anonymous";
386     List<String> groups = new ArrayList<>(securityIdentity.getRoles());
387     String persona = securityIdentity.getAttribute("persona");

389 
390     if (rejection == null || rejection.reason == null || rejection.reason.isBlank()) {
391       return Response.status(400).entity(Map.of("error", "Rejection reason is mandatory")).build();
392     }
393 
394     return accessRequestService.getRequestById(id)
395       .map(r -> {
396         List<String> userGroups = new ArrayList<>(securityIdentity.getRoles());
397         boolean isDesignatedApprover = !"REQUESTER".equals(persona) && r.approverGroups() != null && r.approverGroups().stream().anyMatch(userGroups::contains);
398 
399         boolean isAuthorized = isAdmin || isDesignatedApprover;
400 
401         if (!isAuthorized) {
402           return Response.status(403).entity(Map.of("error", "You are not an authorized approver for this request")).build();
403         }
404 
405         AccessRequest updated = new AccessRequest(
406           r.id(), r.requesterId(), r.userId(), r.principalType(), r.catalogName(), r.schemaName(), r.tableName(),
407           r.resourceType(), r.privileges(), "REJECTED", r.createdAt(), System.currentTimeMillis(),
408           r.justification(), rejection.reason, r.approverGroups(), r.metadata(), r.expirationTime()
409         );
410         accessRequestService.saveRequests(List.of(updated), userId, groups, isAdmin);
411         broadcast("request-updated", List.of(updated));
412         return Response.ok(updated).build();
413       })
414       .orElse(Response.status(404).entity(Map.of("error", "Request not found")).build());
415   }

417   /**
418    * Verifies that the privileges of an approved request have actually been implemented
419    * in the target catalog provider (Drift Verification).
420    *
421    * @param id The request ID
422    * @param securityContext JAX-RS security context
423    * @return Response 200 if verified, 409 if drift detected
424    */
425   @POST
426   @Path("/{id}/verify")
427   @Operation(summary = "Verify access request", description = "Verifies that an approved access request has been implemented in the target catalog")
428   public Response verifyRequest(@PathParam("id") String id, @Context SecurityContext securityContext) {
429     String userId = securityIdentity.getPrincipal() != null ? securityIdentity.getPrincipal().getName() : "anonymous";
430     List<String> groups = new ArrayList<>(securityIdentity.getRoles());
431     String persona = securityIdentity.getAttribute("persona");
432     boolean isAdmin = !"REQUESTER".equals(persona) && (securityIdentity.hasRole("ADMIN") || securityIdentity.hasRole("SECURITY_ADMIN") || securityIdentity.hasRole("APPROVER") || securityIdentity.hasRole("GOVERNANCE_ADMIN") || securityIdentity.hasRole("AUDITOR") || groups.contains("admins"));
433 
434     return accessRequestService.getRequestById(id)
435       .map(r -> {
436         if (!"APPROVED".equals(r.status())) {
437           return Response.status(400).entity(Map.of("error", "Only approved requests can be verified")).build();
438         }
439 
440         String principal = r.userId() != null ? r.userId() : r.requesterId();
441         String path = "/" + r.catalogName() + "/" + r.schemaName() + "/" + r.tableName();
442         boolean verified = catalogService.verifyPolicy(r.catalogName(), path, r.privileges().get(0), principal);
443 
444         if (verified) {
445           AccessRequest verifiedReq = new AccessRequest(
446             r.id(), r.requesterId(), r.userId(), r.principalType(), r.catalogName(), r.schemaName(), r.tableName(),
447             r.resourceType(), r.privileges(), "VERIFIED", r.createdAt(), System.currentTimeMillis(),
448             r.justification(), r.rejectionReason(), r.approverGroups(), r.metadata(), r.expirationTime()
449           );
450           accessRequestService.saveRequests(List.of(verifiedReq), userId, groups, isAdmin);
451           broadcast("request-verified", verifiedReq);
452           return Response.ok(new HalAccessRequest(verifiedReq)).build();
453         } else {
454           return Response.status(409).entity(Map.of("error", "Drift detected: Permissions not yet implemented in target catalog")).build();
455         }
456       })
457       .orElse(Response.status(404).build());
458   }
